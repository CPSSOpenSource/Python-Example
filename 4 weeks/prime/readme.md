# 소수의 합 구하기

```python

# 문제 - 범위 내에 있는 모든 소수의 합 구하기
#
# 주요 조건
# - prime_sum(n)이라는 함수를 만들어서 사용하세요.
# - n의 범위는 0<=n<=1000까지이며, 벗어날 시 에러 메세지를 출력하며 프로그램을 종료합니다.
#
# 핵심 부분
# - 소수를 어떻게 판별한 것인지를 생각해봐야 합니다.
#
# 소스 작성자 - ruskonert@gmail.com

# sys 모듈을 이용하여 프로그램 종료를 하기 위해 import하였습니다.
import sys

# prime_sum이라는 함수를 정의하는데, n이라는 매개변수를 갖고 있습니다.
def prime_sum(n):
    # 모두 합한 값은 최소값이 0입니다. 왜냐하면 소수가 한개도 나오지 않았다면
    # 0이 나오기 떄문입니다.
    sum = 0
    
    # 2부터 n까지 숫자를 사용할 것이고, 이 숫자 하나 하나를 판별할 것입니다.
    # 2부터 시작해 n까지 도달하면서 range_in_number 변수에다 값을 넣어 사용합니다.
    for range_in_number in range(2, n + 1):
        
        # 해당 반복문은 range_in_number이 소수인지 판별하는 것입니다.
        # 소수를 판별하는 방법은 2부터 range_in_number까지 모두 나눠서
        # 만약 나머지가 0이 나오는 게 한 숫자라도 있으면, 해당 range_in_number는
        # 그 숫자로 나누어 떨어지는 수입니다.
        # 즉, 그 숫자는 소수가 아니라는 것입니다.
        # 
        # j는 range_in_number와 나누어 볼 숫자입니다.
        # j는 range_in_number - 1부터 시작해 1씩 감소하면서 1에 도달할 때까지 반복합니다.
        for j in range(range_in_number-1, 0, -1):
            # 만약 j가 1이라면
            # 나누어 떨어지는 숫자가 1개도 없었다는 것입니다.
            # 만약 있었다면, 아래의 break문에 걸려 j가 1이 되기도 전에
            # 해당 for문은 종료했을 것입니다. (위에 있는 for가 아닌)
            if j == 1:
                # 해당 range_in_number는 소수이므로 합을 모두 더한 값을 저장하는
                # 변수에 더해줍니다.
                sum = sum + range_in_number
            
            # 만약 나누고자 하는 수 j로 range_in_number를 나눠봤는데
            # 나머지가 0이면
            if range_in_number % j == 0:
                # 해당 range_in_number는 더이상 진행할 이유가 없습니다.
                # 나머지가 0이라는 것은 소수가 아니라는 뜻이므로
                # 해당 for문을 종료하고, range_in_number에 1을 더해 
                # 다음 숫자를 판별해볼 것입니다.
                break
    # 소수를 합한 값을 반환합니다.            
    return sum

# main 함수입니다.
# 이것은 __main__ 모듈에 의해 실행되었습니다.
def main():
    # 입력을 받고, 그 값에 대해서 메시지를 출력한 후
    # 또 다시 숫자를 입력하는 메세지가 뜹니다.
    # 이것은 예외 처리가 나올 떄까지 계속해서 실행된다는 것입니다.
    while True:
        # input 함수는 기본적으로 str형으로 반환합니다.
        # 따라서 형 변환을 해야 합니다.
        n = int(input("숫자를 입력하세요: "))
        
        # N의 조건에 대한 처리 방법은 다양합니다.
        # N의 범위는 0이상 1000이하인데, 이것은 해당 범위를 벗어났을 때에 대한 경우입니다.
        if n < 0 or n > 1000:
            print("범위를 벗어났습니다. 프로그램을 종료합니다.")
            sys.exit(0)
         
        # 만약에 n이 0이나 1이면
        # 구할 숫자가 없습니다. 따라서 출력 결과를 고려해봤을 때
        # 해당 메세지를 출력한 후, 다시 입력하는 
        if n == 0 or n == 1:
            print("더할 숫자가 없습니다!")
            continue
            
        # 위에 있는 여러가지의 예외 처리를 모두 넘어갔습니다.
        # 이것은 N은 정상적인 범위에 있다는 것을 의미합니다.
        print("2부터 %d까지의 모든 소수 합은 %d입니다." % (n, prime_sum(n)))


if __name__== "__main__":
    main()
```
